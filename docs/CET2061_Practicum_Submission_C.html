# Define the function f
def calculate_f(CT, SP, ST, R, MS, MWS, NR, A):
    term1 = (np.exp(CT + 1) / np.log2((SP + 5) / 3)) + (ST + 1) / (R + 1)
    term2 = MS * MWS / (NR + 1)
    f = term1 * MS + term2 - A
    return f
